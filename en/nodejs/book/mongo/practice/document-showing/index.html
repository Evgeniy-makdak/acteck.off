Showing a document from the database

Let's make a route that will display a specific user in the browser. Let the name of the user we want to display be specified by the route parameter:

app.get('/user/show/:name', function(req, res) { console.log(req.params.name); });

Let's get the user by his name:

app.get('/user/show/:name', function(req, res) { let name = req.params.name; let user = await coll.findOne({name: name}); console.log(user); });

Let's send the received user's data to the browser:

app.get('/user/show/:name', function(req, res) { let name = req.params.name; let user = await coll.findOne({name: name}); res.send('name: ' + user.name + 'salary: ' + user.salary); });
