Route options and 404 error in Express

Let us have a route whose parameter is a certain number:

app.get('/test/:num/', function(req, res) { res.send(req.params.num); });

Let's only allow addresses with numbers from1to9, and for the remaining addresses we will give404error:

app.get('/test/:num/', function(req, res) { let num = req.params.num; if (num >= 1 && num <= 9) { res.send('your num: ' + num); } else { res.status(404).send('not found'); } });

Now let's check whether a number is entered or not using a regular expression:

app.get('/test/:num/', function(req, res) { let num = req.params.num; if (/\d+/.test(num)) { res.send('your num: ' + num); } else { res.status(404).send('not found'); } });

Make a route with some parameter. Implement a parameter check to ensure that this is a date in the formatyear-month-day. Return404if this is not the case.

General404error

In the above code it will be that if the address starts with/test/and then not a number, it will fly to the browser404error. But an arbitrary URL that does not start with/test/. We must handle such cases throughapp.use:

app.use(function(req, res) { res.status(404).send('not found'); });

The complete code will look like this:

app.get('/test/:num/', function(req, res) { let num = req.params.num; if (/\d+/.test(num)) { res.send('your num: ' + num); } else { res.status(404).send('not found'); // начало с /test/, но потом не число } }); app.use(function(req, res) { res.status(404).send('not found'); // адреса, не начинающиеся с /test/ });
