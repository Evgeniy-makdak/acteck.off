Synchronous style fetch in JavaScript

You can also work withfetchVsynchronized style:

button.addEventListener('click', async function() { let response = await fetch('/ajax.html'); let text = await response.text(); console.log(text); });

Rewrite the following code in synchronous style:

button.addEventListener('click', function() { let promise = fetch('/ajax.html') .then( response => { if (response.ok) { return response.text(); } else { throw new Error('плохой статус ответа'); } }, ).then( text => { console.log(text); } ).catch( error => { console.log(error); } ); });
