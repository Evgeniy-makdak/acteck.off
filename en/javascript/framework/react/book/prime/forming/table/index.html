Outputting an array of objects as an HTML table in React

Let us again have our array with products:

const prods = [ {id: 1, name: 'product1', cost: 100}, {id: 2, name: 'product2', cost: 200}, {id: 3, name: 'product3', cost: 300}, ];

Let's display the elements of our array as an HTML table:

function App() { const rows = prods.map(function(item) { return <tr key={item.id}> <td>{item.name}</td> <td>{item.cost}</td> </tr>; }); return <table> <tbody> {rows} </tbody> </table>; }

Let's add column headings to our table:

function App() { const rows = prods.map(function(item) { return <tr key={item.id}> <td>{item.name}</td> <td>{item.cost}</td> </tr>; }); return <table> <thead> <tr> <td>название</td> <td>стоимость</td> </tr> </thead> <tbody> {rows} </tbody> </table>; }
