Array element zero and all others in ES6

Let us be given an array and we want to separate the zero element of this array and the remaining ones. Destructuring and the operator will help us with thisrest.

Using this ES6 trick, let's correct the function that finds the intersection of arrays, which I talked aboutin the next video.

function getInt(...arrs) { let result = []; let [arr0, ...rest] = arrs; for (let elem of arr0) { if(inArrays(rest, elem)) { result.push(elem); } } return result; } function inArrays(arrs, elem) { for (let arr of arrs) { if (!inArray(arr, elem)) { return false; } } return true; } function inArray(arr, elem) { return arr.find(e => e === elem); } console.log(getInt([1, 2, 3], [2, 3, 4, 5], [2, 3, 4, 6]));
