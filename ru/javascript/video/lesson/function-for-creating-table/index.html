Функция для создания HTML таблицы на JavaScript

В данном уроке я покажу реализацию функции для создания HTML таблицы на JavaScript. В функции задается количество колонок и строк таблицы.

Исходники кода

<div id="elem"></div> table td { padding: 20px; border: 1px solid black; } var elem = document.querySelector('#elem'); createTable(elem, 3, 3); function createTable(parent, cols, rows) { var table = document.createElement('table'); for (var i = 0; i < rows; i++) { var tr = document.createElement('tr'); for (var j = 0; j < cols; j++) { var td = document.createElement('td'); tr.appendChild(td); } table.appendChild(tr); } parent.appendChild(table); }
