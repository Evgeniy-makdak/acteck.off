Создание HTML таблицы из двухмерного массива

В данном уроке я расскажу вам, как создать HTML таблицу из двухмерного массива, на основе которого будут строится колонки и ряды таблицы.

Исходники кода

var table = document.querySelector('#table'); var arr = [[1, 2, 3], [4, 5, 6], [7, 8, 9], [7, 8, 9]]; fillTable(table, arr); function fillTable(table, arr) { for (var i = 0; i < arr.length; i++) { var tr = document.createElement('tr'); for (var j = 0; j < arr[i].length; j++) { var td = document.createElement('td'); td.innerHTML = arr[i][j]; tr.appendChild(td); } table.appendChild(tr); } }
